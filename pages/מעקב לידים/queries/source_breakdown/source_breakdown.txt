SELECT "שם_שותף" AS מקור, COUNT(*) AS לידים, COUNT(CASE WHEN "מינוי_סוכן" > 0 THEN 1 END) AS מינוי_סוכן, COUNT(CASE WHEN "סגירות" > 0 THEN 1 END) AS סגירות, COALESCE(SUM("מינוי_סוכן"), 0) AS סכום_מינוי_סוכן, COALESCE(SUM("סגירות"), 0) AS סכום_סגירות, CASE WHEN COUNT(*) > 0 THEN ROUND(COUNT(CASE WHEN "מינוי_סוכן" > 0 THEN 1 END)::numeric / COUNT(*)::numeric * 100, 1) ELSE 0 END AS אחוז_מינוי, CASE WHEN COUNT(*) > 0 THEN ROUND(COUNT(CASE WHEN "סגירות" > 0 THEN 1 END)::numeric / COUNT(*)::numeric * 100, 1) ELSE 0 END AS אחוז_סגירות FROM partner_commissions_view WHERE ('{{Select_Month.selectedOptionValue}}' = '' OR TO_CHAR("תאריך_עדכון", 'MM/YYYY') = '{{Select_Month.selectedOptionValue}}') GROUP BY "שם_שותף" ORDER BY לידים DESC
