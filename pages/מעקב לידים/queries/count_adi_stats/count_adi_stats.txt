SELECT COUNT(*) AS total, COUNT(CASE WHEN "חברת_ביטוח" ILIKE '%פניקס%' THEN 1 END) AS פניקס, COUNT(CASE WHEN "חברת_ביטוח" = 'הראל' THEN 1 END) AS הראל, COUNT(CASE WHEN "חברת_ביטוח" = 'כלל' THEN 1 END) AS כלל, COUNT(CASE WHEN "חברת_ביטוח" ILIKE '%פניקס%' AND "סגירות" IS NOT NULL THEN 1 END) AS פניקס_סגירות, COUNT(CASE WHEN "חברת_ביטוח" = 'הראל' AND "סגירות" IS NOT NULL THEN 1 END) AS הראל_סגירות, COUNT(CASE WHEN "חברת_ביטוח" = 'כלל' AND "סגירות" IS NOT NULL THEN 1 END) AS כלל_סגירות, COALESCE(SUM("מינוי_סוכן"), 0) AS סהכ_מינוי, COALESCE(SUM("סגירות"), 0) AS סהכ_סגירה FROM partner_commissions_view WHERE ('{{Select_Month.selectedOptionValue}}' = '' OR TO_CHAR("תאריך_עדכון", 'MM/YYYY') = '{{Select_Month.selectedOptionValue}}') AND ('{{Select_Partner.selectedOptionValue}}' = '' OR "שם_שותף" = '{{Select_Partner.selectedOptionValue}}')
