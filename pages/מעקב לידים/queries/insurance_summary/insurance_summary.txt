SELECT "חברת_ביטוח" AS חברה, COUNT(*) AS לידים, COUNT(CASE WHEN "סגירות" > 0 THEN 1 END) AS סגירות, ROUND(SUM(COALESCE("סגירות", 0))) AS סכום_סגירות, CASE WHEN COUNT(*) > 0 THEN ROUND(COUNT(CASE WHEN "סגירות" > 0 THEN 1 END)::numeric / COUNT(*)::numeric * 100, 1) ELSE 0 END AS אחוז_סגירה FROM partner_commissions_view WHERE "חברת_ביטוח" IS NOT NULL AND "חברת_ביטוח" != '' AND ('{{Select_Month.selectedOptionValue}}' = '' OR TO_CHAR("תאריך_עדכון", 'MM/YYYY') = '{{Select_Month.selectedOptionValue}}') AND ('{{Select_Partner.selectedOptionValue}}' = '' OR "שם_שותף" = '{{Select_Partner.selectedOptionValue}}') GROUP BY "חברת_ביטוח" ORDER BY סגירות DESC
