SELECT
    p.name AS מקור,
    COUNT(*) AS לידים,
    COUNT(CASE WHEN l.agent_appointment_amount > 0 THEN 1 END) AS מינוי_סוכן,
    COUNT(CASE WHEN l.closing_amount > 0 THEN 1 END) AS סגירות,
    COALESCE(SUM(l.closing_amount), 0) AS סכום_סגירות,
    COALESCE(SUM(l.agent_appointment_amount), 0) AS סכום_מינוי,
    CASE WHEN COUNT(*) > 0 THEN ROUND(COUNT(CASE WHEN l.agent_appointment_amount > 0 THEN 1 END)::numeric / COUNT(*)::numeric * 100, 1) ELSE 0 END AS אחוז_מינוי,
    CASE WHEN COUNT(*) > 0 THEN ROUND(COUNT(CASE WHEN l.closing_amount > 0 THEN 1 END)::numeric / COUNT(*)::numeric * 100, 1) ELSE 0 END AS אחוז_סגירות
FROM leads l LEFT JOIN partners p ON l.partner_id = p.id
WHERE p.name IS NOT NULL
AND ('{{Select_Period.selectedOptionValue}}' = 'all'
    OR ('{{Select_Period.selectedOptionValue}}' = 'current_month' AND l.created_at >= DATE_TRUNC('month', CURRENT_DATE))
    OR ('{{Select_Period.selectedOptionValue}}' = 'last_month' AND l.created_at >= DATE_TRUNC('month', CURRENT_DATE - INTERVAL '1 month') AND l.created_at < DATE_TRUNC('month', CURRENT_DATE))
    OR ('{{Select_Period.selectedOptionValue}}' = 'ytd' AND l.created_at >= DATE_TRUNC('year', CURRENT_DATE)))
GROUP BY p.name ORDER BY COUNT(*) DESC
